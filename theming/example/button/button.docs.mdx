import { Button } from './button';
import { Theme } from '@learn-bit-react/theming.themes.theme-provider';
import { LightTheme } from '@learn-bit-react/theming.themes.light-theme';
import { DarkTheme } from '@learn-bit-react/theming.themes.dark-theme';

import {
  PinkTheme,
  pinkTheme,
} from '@learn-bit-react/theming.themes.pink-theme';

A Button component to show theming usage.

### Wrap the component in a ThemeProvider to apply the theme

```js
<Theme.ThemeProvider>
  <Button>Default Theme</Button>
</Theme.ThemeProvider>
```

### Override the default theme with a pink theme

```js
<Theme.ThemeProvider overrides={pinkTheme}>
  <Button>Pink Theme</Button>
</Theme.ThemeProvider>
```

### Override the default theme with a pink theme

This works the same as the example above

```js
<PinkTheme>
  <Button>Pink Theme</Button>
</PinkTheme>
```

### Render button with different themes

```js
<Theme.ThemeProvider>
  <Button>Default Theme</Button>

  <PinkTheme>
    <Button>Pink Theme</Button>
  </PinkTheme>
</Theme.ThemeProvider>
```
